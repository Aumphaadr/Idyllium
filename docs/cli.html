Структура программы

Каждая программа начинается с функции main():

use console;

main() {
    console.write("Hello, World!", '\n');
}

Тип возвращаемого значения не указывается. Аргументы для main не требуются.

Типы данных и переменные

  • int — целое число
  • float — число с плавающей точкой
  • char — одиночный символ
  • string — строка
  • bool — логическое значение
  • array<T, N> — массив фиксированной длины
  • dyn_array<T> — динамический массив
  • void — для функций без возвращаемого значения

Пример объявления переменных:

main() {
    int age = 18;
    float pi = 3.14;
    char letter = 'Q';
    string name = "Alice";
    bool is_active = true;
}

Консольный ввод/вывод

Для работы с консолью используется встроенная библиотека console:

use console;

main() {
    int a;
    float b;
    string name;

    console.write("Введите имя");
    name = console.get_string();

    console.write("Введите целое число");
    a = console.get_int();

    console.write("Введите дробное число");
    b = console.get_float();

    console.write("Привет, ", name, "! Ваши числа: ", a, " и ", b, '\n');
}

В языке Idyllium чёткие типы и читаемые функции для ввода/вывода в любом желаемом типе.

Никаких Int32.TryParse() и int(input('Введи число: '))

А ещё console.write() не делает перенос строки без явного вывода символа '\n'. Красота!

Арифметика

Все арифметические операции в Idyllium предсказуемы и однозначно типизируемы:

main() {
    int a = 9;
    int b = 4;
    float d = 2.5;

    // базовые операции
    int r1 = a + b; // 13
    int r2 = a - b; // 5
    int r3 = a * b; // 36
    float r4 = a / b; // 2.25

    // взаимодействие 'int' и 'float' всегда порождает 'float'
    float r5 = a + d; // 11.5
    float r6 = b * d; // 10.0

    // неявное преобразование допустимо только от 'int' к 'float'
    float f = 3; // 3.0
    int n = 7.5; // TypeError

    // для целочисленного деления используются функции 'div()' и 'mod()'
    int quotient = div(43, 2); // 21
    int remainder = mod(43, 2); // 1
}

Оператор % намеренно удалён, чтобы 50 % 14 никто не перепутал с "50 процентов от 14". С заботой о детях!

Явные преобразования

Для преобразования данных от одного типа к другому используются функции:

main() {
    // преобразования чисел в строки
    string s1 = to_string(25); // "25"
    string s2 = to_string(4.7); // "4.7"

    // преобразования строк в числа
    int n1 = to_int("1453"); // 1453
    float f1 = to_float("36.4"); // 36.4

    // преобразования между числовыми типами
    int n2 = to_int(9.63); // 9
    float f2 = to_float(15); // 15.0
}

Конкатенация

В языке Idyllium конкатенация возможна только в случаях:

  • string + string
  • string + char
  • char + string
use console;

main() {
    string greeting = "Hello, ";
    string name = "Alice";
    char mark = '!';

    string r1 = greeting + name; // 'string' + 'string'
    console.write(r1, '\n'); // "Hello, Alice"

    string r2 = r1 + mark; // 'string' + 'char'
    console.write(r2, '\n'); // "Hello, Alice!"
}

Все остальные типы данных не поддерживают конкатенацию без явного преобразования в строку.

main() {
    // конкатенация строк с числами невозможна
    string bad = "Year: " + 2025; // TypeError: '+' is unable to 'str' and 'int'

    // но если число преобразовать в строку...
    string good = "Year: " + to_string(2025); // "Year: 2025"
}

Условный оператор

Реализация стандартна, как в C-подобных языках:

use console;

main() {
    int a = console.get_int();
    int b = console.get_int();

    if (a > b)
    {
        console.write(a, " больше, чем ", b, '\n');
    }
    else if (a == b)
    {
        console.write(a, " равно ", b, '\n');
    }
    else
    {
        console.write(a, " меньше, чем ", b, '\n');
    }
}

Операторы циклов

Также аналогичны C-подобным языкам:

use console;

main() {
    // цикл 'for'
    for (int i = 0; i < 20; i = i + 1)
    {
        console.write("i = ", i, '\n');
    }

    // цикл 'while'
    int q = 1;
    while (q < 500)
    {
        console.write(q, '\n');
        q = q * 2;
    }

    // цикл 'do-while'
    int right_answer = 71;
    int your_answer;
    do
    {
        console.write("27 + 44 = ");
        your_answer = console.get_int();
    } while (your_answer != right_answer);
}

Псевдослучайные числа

Для генерации псевдослучайных чисел используется встроенная библиотека random:

use console;
use random;

main() {
    int n = random.create_int(10, 99); // границы включены
    float d = random.create_float(2.5, 4);

    // получение случайного элемента из строки либо массива
    string alphabet = "abcdefghijklmn";
    char ch = random.choose_from(alphabet);
}

Да, всё настолько просто! Никаких жутких колдунств вроде 10 + rand() % 90! Это ли не прекрасно?

Статические массивы

Реализация массивов в Idyllium выглядит следующим образом:

use console;
use random;

main() {
    array<int, 5> small = [11, 22, 33, 44, 55]; // массив на 5 ячеек типа 'int'
    console.write(small, '\n'); // [11, 22, 33, 44, 55]

    array<int, 50> big; // массив на 50 ячеек типа 'int'
    
    for (int i = 0; i < 50; i = i + 1)
    {
        big[i] = random.create_int(100, 999);
        console.write("big[", i, "]: ", big[i], '\n');
    }

    // попытка выйти за границы массива = ошибка
    console.write(big[60], '\n'); // IndexError: array index out of bounds
}

Нумерация ячеек массива начинается с нуля, как во всех нормальных языках.

Ячейки массива строго однотипны. Положить в массив число, строку и кота — не в Idyllium, дружище.

Да и зачем? Кто в здравом уме вообще так делает?

Динамические массивы

Мы в Idyllium знаем, что динамические массивы логично изучать после статических.

Поэтому мы сделали их похожими друг на друга, чтобы не переворачиваться с ног на голову при их изучении!

use console;

main() {
    dyn_array<int> nums; // динамический массив типа 'int'

    // добавление ячеек в конец
    nums.add(42);
    nums.add(74);
    nums.add(13);
    nums.add(29);
    nums.add(81);

    console.write(nums, '\n'); // [42, 74, 13, 29, 81]

    // удаление ячейки по индексу
    nums.remove_at(1);

    console.write(nums, '\n'); // [42, 13, 29, 81]

    // получение количества ячеек в массиве
    int L = nums.length(); // 4
}

Вы уже чувствуете, насколько стали ненужными комментарии?

Но мы всё равно их оставим. Вдруг какому-нибудь "хлебушку" что-то будет непонятно.

Использование библиотек

Мы знаем, что "пространство имён" — это термин, который невозможно объяснить человеку, который ни разу не создавал самописные библиотеки и не сталкивался с конфликтами имён.

Поэтому вызов команд в Idyllium производится по схеме библиотека.команда():

use console;
use math;
use time;

main() {
    console.write("Подождите, округляю число...", '\n');

    time.sleep(2.5); // пауза на 2.5 секунды
    float pi4 = math.round(3.1415926535, 4); // округление до 4 знаков после точки

    console.write("Округлено: ", pi4, '\n'); // Округлено: 3.1416
}

Процедуры и функции

Синтаксис функций аналогичен C-подобным языкам, но с небольшими изменениями:

use console;

// описание void-функции (процедуры)
void function draw_line(int L)
{
    string res = "";
    for (int i = 0; i < L; i = i + 1)
    {
        res = res + '-';
    }
    console.write(res, '\n');
}

main() {
    // вызов void-функции (процедуры)
    draw_line(5); // выведется "-----"
    draw_line(10); // выведется "----------"
}

Функции с возвратом значения выглядят схожим образом:

// описание int-функции
int function cube(int N)
{
    return N * N * N;
}

main() {
    // вызов int-функции
    int a = cube(2); // 8
    int b = cube(5); // 125
    int c = cube(7); // 343
}

Работа с файлами

Запись данных в текстовый файл:

use file;

main() {
    file.ostream fout = file.open("output.txt", "write");
    fout.write_line("Привет, мир!", '\n');
    fout.write_line("Сейчас ", 2025, " год", '\n');
    fout.write_line("Файл создан с помощью Idyllium!");
    fout.close();
}

Чтение данных из текстового файла:

use console;
use file;

main() {
    file.istream fin = file.open("input.txt", "read");
    while (fin.has_next_line())
    {
        string one_line = fin.read_line();
        console.write(one_line, '\n');
    }
    fin.close();
}
